// --- SECRET COVER PAGE (මුහුණත) ---
class SecretCoverPage extends StatefulWidget {
  const SecretCoverPage({super.key});
  @override
  State<SecretCoverPage> createState() => _SecretCoverPageState();
}

// WidgetsBindingObserver භාවිතයෙන් Auto-lock විශේෂාංගය පාලනය කරයි
class _SecretCoverPageState extends State<SecretCoverPage> with WidgetsBindingObserver {
  int _tapCount = 0;
  Timer? _tapTimer;

  @override
  void initState() {
    super.initState();
    WidgetsBinding.instance.addObserver(this);
  }

  @override
  void dispose() {
    WidgetsBinding.instance.removeObserver(this);
    super.dispose();
  }

  @override
  void didChangeAppLifecycleState(AppLifecycleState state) {
    // ඇප් එක Background ගිය සැනින් පියවා මුල් පිටුවට (Cover) යොමු කරයි
    if (state == AppLifecycleState.paused || state == AppLifecycleState.inactive) {
      Navigator.of(context).popUntil((route) => route.isFirst);
    }
  }

  void _handleTap() {
    _tapCount++;
    if (_tapCount == 3) {
      _tapTimer?.cancel();
      _tapCount = 0;
      Navigator.push(context, MaterialPageRoute(builder: (context) => const VaultSecurityHandler()));
    } else {
      _tapTimer?.cancel();
      _tapTimer = Timer(const Duration(milliseconds: 500), () => _tapCount = 0);
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        leading: const Icon(Icons.photo_library, color: Colors.blue),
        title: const Text("My Memory", style: TextStyle(fontWeight: FontWeight.bold)),
        centerTitle: false,
        elevation: 0,
        backgroundColor: Colors.transparent,
      ),
      extendBodyBehindAppBar: true,
      body: GridView.builder(
        padding: const EdgeInsets.only(top: 100, left: 10, right: 10, bottom: 10),
        gridDelegate: const SliverGridDelegateWithFixedCrossAxisCount(crossAxisCount: 2, crossAxisSpacing: 12, mainAxisSpacing: 12),
        itemCount: 20,
        itemBuilder: (context, index) => GestureDetector(
          onTap: _handleTap,
          child: ClipRRect(
            borderRadius: BorderRadius.circular(15),
            child: Image.network("https://picsum.photos/400/600?random=$index", fit: BoxFit.cover),
          ),
        ),
      ),
    );
  }
}

// --- SECURITY HANDLER (PIN & BIOMETRICS) ---
class VaultSecurityHandler extends StatefulWidget {
  const VaultSecurityHandler({super.key});
  @override
  _VaultSecurityHandlerState createState() => _VaultSecurityHandlerState();
}

class _VaultSecurityHandlerState extends State<VaultSecurityHandler> {
  final LocalAuthentication auth = LocalAuthentication();
  String _inputPin = "";
  Timer? _lockoutTimer;
  String _remainingTime = "";

  // සැබෑ Biometric Authentication (Fingerprint/FaceID)
  Future<void> _authenticateBiometrics() async {
    try {
      bool authenticated = await auth.authenticate(
        localizedReason: 'Please authenticate to open the vault',
        options: const AuthenticationOptions(biometricOnly: true, stickyAuth: true),
      );
      if (authenticated) {
        Navigator.pushReplacement(context, MaterialPageRoute(builder: (context) => const MainVault(isDecoy: false)));
      }
    } catch (e) {
      _showMsg("Biometric error or not set up.");
    }
  }

  void _onKeyPress(String value) {
    if (lockoutEndTime != null) return;
    setState(() {
      if (value == "BACK") {
        if (_inputPin.isNotEmpty) _inputPin = _inputPin.substring(0, _inputPin.length - 1);
      } else if (value == "BIO") {
        _authenticateBiometrics();
      } else if (_inputPin.length < 4) {
        _inputPin += value;
        if (_inputPin.length == 4) {
          Future.delayed(const Duration(milliseconds: 200), () => _processPin());
        }
      }
    });
  }

  void _processPin() {
    String input = _inputPin;
    setState(() => _inputPin = ""); 
    
    if (masterPin == null || decoyPin == null) {
      if (setupStep == 1) { tempMasterPin = input; setupStep = 2; _showMsg("Confirm Master PIN"); }
      else if (setupStep == 2) {
        if (input == tempMasterPin) { masterPin = input; setupStep = 3; _showMsg("Set Decoy PIN"); }
        else { setupStep = 1; _showMsg("Mismatch! Try again."); }
      }
      else if (setupStep == 3) {
        if (input == masterPin) { _showMsg("Decoy PIN must be different!"); }
        else { decoyPin = input; _showMsg("Setup Successful!"); Navigator.pop(context); }
      }
    } else {
      if (input == masterPin) {
        wrongAttempts = 0;
        Navigator.pushReplacement(context, MaterialPageRoute(builder: (context) => const MainVault(isDecoy: false)));
      } else if (input == decoyPin) {
        wrongAttempts = 0;
        Navigator.pushReplacement(context, MaterialPageRoute(builder: (context) => const MainVault(isDecoy: true)));
      } else {
        _handleSilentFailure();
      }
    }
  }

  void _handleSilentFailure() {
    wrongAttempts++;
    if (wrongAttempts >= 5) {
      setState(() { lockoutEndTime = DateTime.now().add(const Duration(minutes: 5)); _startTimer(); });
    } else {
      _showMsg("Incorrect PIN! ($wrongAttempts/5)");
    }
  }

  void _startTimer() {
    _lockoutTimer = Timer.periodic(const Duration(seconds: 1), (timer) {
      final now = DateTime.now();
      if (lockoutEndTime == null || now.isAfter(lockoutEndTime!)) {
        timer.cancel();
        setState(() { lockoutEndTime = null; wrongAttempts = 0; });
      } else {
        final diff = lockoutEndTime!.difference(now);
        setState(() { _remainingTime = "${diff.inMinutes}:${(diff.inSeconds % 60).toString().padLeft(2, '0')}"; });
      }
    });
  }

  void _showMsg(String m) => ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text(m), duration: const Duration(seconds: 1)));

  Widget _buildNumBtn(String val) {
    bool isBack = val == "BACK";
    return GestureDetector(
      onTap: () => _onKeyPress(val),
      child: Container(
        margin: const EdgeInsets.all(10),
        height: 70, width: 70,
        decoration: BoxDecoration(
          shape: BoxShape.circle,
          color: Colors.white.withOpacity(0.05),
          border: Border.all(color: Colors.cyanAccent.withOpacity(0.3)),
        ),
        child: Center(
          child: isBack 
            ? const Icon(Icons.backspace_outlined, color: Colors.white, size: 24)
            : Text(val, style: const TextStyle(fontSize: 24, fontWeight: FontWeight.bold, color: Colors.white))
        ),
      ),
    );
  }

  @override
  Widget build(BuildContext context) {
    bool isLocked = lockoutEndTime != null;
    String currentLabel = masterPin == null ? (setupStep == 1 ? "Master PIN" : "Confirm Master PIN") : (decoyPin == null ? "Decoy PIN" : "Enter PIN");

    return Scaffold(
      body: Container(
        decoration: const BoxDecoration(
          gradient: LinearGradient(begin: Alignment.topLeft, end: Alignment.bottomRight, colors: [Color(0xFF0F172A), Color(0xFF1E293B)]),
        ),
        child: SafeArea(
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              Icon(isLocked ? Icons.lock_clock : Icons.shield_moon, size: 80, color: Colors.cyanAccent),
              const SizedBox(height: 20),
              Text(isLocked ? "System Locked: $_remainingTime" : currentLabel, style: const TextStyle(fontSize: 22, color: Colors.white, fontWeight: FontWeight.w300)),
              const SizedBox(height: 40),
              Row(mainAxisAlignment: MainAxisAlignment.center, children: List.generate(4, (i) => Container(margin: const EdgeInsets.all(8), width: 15, height: 15, decoration: BoxDecoration(shape: BoxShape.circle, color: i < _inputPin.length ? Colors.cyanAccent : Colors.white24)))),
              const SizedBox(height: 40),
              if (!isLocked) ...[
                Column(children: [
                  Row(mainAxisAlignment: MainAxisAlignment.center, children: ["1", "2", "3"].map(_buildNumBtn).toList()),
                  Row(mainAxisAlignment: MainAxisAlignment.center, children: ["4", "5", "6"].map(_buildNumBtn).toList()),
                  Row(mainAxisAlignment: MainAxisAlignment.center, children: ["7", "8", "9"].map(_buildNumBtn).toList()),
                  Row(mainAxisAlignment: MainAxisAlignment.center, children: [
                    const SizedBox(width: 90), 
                    _buildNumBtn("0"),
                    _buildNumBtn("BACK"),
                  ]),
                ]),
                const SizedBox(height: 20),
                GestureDetector(
                  onTap: () => _onKeyPress("BIO"),
                  child: Column(
                    children: [
                      Icon(Icons.fingerprint, color: Colors.cyanAccent.withOpacity(0.8), size: 45),
                      const SizedBox(height: 5),
                      const Text("Unlock with Biometrics", style: TextStyle(color: Colors.white54, fontSize: 12)),
                    ],
                  ),
                ),
              ],
            ],
          ),
        ),
      ),
    );
  }
}
